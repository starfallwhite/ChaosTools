【2】实现注册器

MODEL_REGISTRY = {}

def register_model(model):
    MODEL_REGISTRY[model.name] = model

def get_model(name: str):
    return MODEL_REGISTRY.get(name)

def list_models():
    return list(MODEL_REGISTRY.keys())
